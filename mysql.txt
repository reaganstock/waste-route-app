-- Create a single, comprehensive route handling function
CREATE OR REPLACE FUNCTION public.handle_route_changes()
RETURNS TRIGGER AS $$
BEGIN
    -- Update timestamp
    NEW.updated_at = now();
    
    -- Handle efficiency calculation
    IF NEW.total_houses > 0 AND NEW.completed_houses IS NOT NULL THEN
        NEW.efficiency = (NEW.completed_houses::numeric / NEW.total_houses::numeric) * 100;
        -- Calculate completion rate
        NEW.completion_rate = NEW.efficiency;
        -- Calculate houses per hour if duration exists
        IF NEW.duration > 0 THEN
            NEW.houses_per_hour = (NEW.completed_houses::numeric / (NEW.duration / 60));
        END IF;
    ELSE
        NEW.efficiency = 0;
        NEW.completion_rate = 0;
    END IF;

    -- Handle expired routes
    IF NEW.status = 'pending' AND NEW.date < CURRENT_DATE THEN
        NEW.status = 'expired';
    END IF;

    -- Count special houses
    IF TG_OP = 'UPDATE' THEN
        SELECT COUNT(*) INTO NEW.special_houses
        FROM houses
        WHERE route_id = NEW.id 
        AND (status = 'skip' OR status = 'new customer' OR notes IS NOT NULL);
    END IF;

    RETURN NEW;
EXCEPTION
    WHEN OTHERS THEN
        -- On any error, set defaults and continue
        NEW.efficiency = 0;
        NEW.completion_rate = 0;
        NEW.houses_per_hour = 0;
        RETURN NEW;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;
